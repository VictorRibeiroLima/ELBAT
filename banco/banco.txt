--TABELA DE LINGUAS
CREATE TABLE T_AM_LINGUA
  (
    CD_LINGUA  NUMBER (1) NOT NULL ,
    DS_LINGUA  VARCHAR2 (10 CHAR) NOT NULL ,
    SGL_LINGUA CHAR (3 CHAR) NOT NULL
  ) ;
ALTER TABLE T_AM_LINGUA ADD CONSTRAINT CK_LINGUAS_DSLINGV CHECK (DS_LINGUA='INGLES' OR DS_LINGUA='PORTUGUES') ;
ALTER TABLE T_AM_LINGUA ADD CONSTRAINT CK_LINGUAS_SGLLINGV CHECK (SGL_LINGUA='ING' OR SGL_LINGUA='PRT') ;
ALTER TABLE T_AM_LINGUA ADD CONSTRAINT UN_AM_LINGUA_DS UNIQUE ( DS_LINGUA ) ;
ALTER TABLE T_AM_LINGUA ADD CONSTRAINT UN_AM_LINGUA_SGL UNIQUE ( SGL_LINGUA ) ;
ALTER TABLE T_AM_LINGUA ADD CONSTRAINT PK_AM_LINGUA PRIMARY KEY ( CD_LINGUA ) ;
--TABELA DE PERGUNTAS SECRETAS
CREATE TABLE T_AM_PERGUNTA (
    CD_PERGUNTA   NUMBER(1) NOT NULL,
    DS_PERGUNTA   VARCHAR2(80 CHAR) NOT NULL
);
ALTER TABLE T_AM_PERGUNTA ADD CONSTRAINT UN_AM_PERGUNTA UNIQUE ( DS_PERGUNTA ) ;
ALTER TABLE T_AM_PERGUNTA ADD CONSTRAINT PK_AM_PERGUNTA PRIMARY KEY ( CD_PERGUNTA );
--TABELA DE DIFICULDADE
CREATE TABLE t_am_dificuldade (
    cd_dificuldade   NUMBER(1) NOT NULL,
    ds_dificuldade   VARCHAR2(16 CHAR) NOT NULL
);
ALTER TABLE T_AM_DIFICULDADE ADD CONSTRAINT UN_AM_DIFICULDADE_ UNIQUE ( DS_DIFICULDADE ) ;
ALTER TABLE t_am_dificuldade ADD CONSTRAINT pk_am_dificuldade PRIMARY KEY ( cd_dificuldade );
ALTER TABLE t_am_dificuldade
    ADD CONSTRAINT ck_am_dsdificuldade CHECK (
        ds_dificuldade = 'BASICO'
        OR ds_dificuldade = 'INTERMEDIARIO'
        OR ds_dificuldade = 'AVANCADO'
    );
--TABELA DE USUARIO
CREATE TABLE T_AM_USUARIO (
    CD_USUARIO          NUMBER(5) NOT NULL,
    CD_PERGUNTA         NUMBER(1) NOT NULL,
    DS_EMAIL            VARCHAR2(100 CHAR) NOT NULL,
    SGL_GENERO          CHAR(1 CHAR) NOT NULL,
    DS_NOME             VARCHAR2(60 CHAR) NOT NULL,
    DS_APELIDO          VARCHAR2(40 CHAR) NOT NULL,
    DS_SENHA            VARCHAR2(10 CHAR) NOT NULL,
    DS_RESP_PERGUNTA   VARCHAR2(50 CHAR) NOT NULL,
    ADM_SISTEMA         CHAR(1) NOT NULL
);
ALTER TABLE T_AM_USUARIO ADD constraint CK_USUARIO_EMAILA CHECK ( INSTR(DS_EMAIL,'@',1,1) > 0);
ALTER TABLE T_AM_USUARIO ADD CONSTRAINT CK_USUARIO_SENHAL CHECK (LENGTH(DS_SENHA) > 2 );
ALTER TABLE T_AM_USUARIO ADD CONSTRAINT CK_USUARIO_NOMEL CHECK (LENGTH(DS_NOME)>2);
ALTER TABLE T_AM_USUARIO ADD CONSTRAINT CK_USUARIO_GENEROV CHECK (SGL_GENERO = 'F' OR SGL_GENERO = 'M' OR SGL_GENERO = 'I');
ALTER TABLE T_AM_USUARIO ADD CONSTRAINT CK_USUARIO_ADMSYS CHECK( ADM_SISTEMA ='1' OR ADM_SISTEMA='0' OR ADM_SISTEMA='2');
ALTER TABLE T_AM_USUARIO ADD CONSTRAINT PK_AM_USUARIO PRIMARY KEY ( CD_USUARIO );
ALTER TABLE T_AM_USUARIO ADD CONSTRAINT CK_USUARIO_UNIQUE_APELIDO UNIQUE ( DS_APELIDO );
ALTER TABLE T_AM_USUARIO ADD CONSTRAINT FK_AM_USUARIO_PERGUNTA FOREIGN KEY ( CD_PERGUNTA )REFERENCES T_AM_PERGUNTA ( CD_PERGUNTA );
alter table t_am_usuario add constraint CK_USUARIO_UNIQUE_EMAIL UNIQUE (DS_EMAIL);
--TABELA DE CONFIG DO USUARIO
CREATE TABLE t_am_config_usuario (
    cd_config           NUMBER(1) NOT NULL,
    cd_usuario          NUMBER(5) NOT NULL,
    cd_lingua           NUMBER(1) NOT NULL,
    nr_opcao_receberm   NUMBER(1) NOT NULL,
    cd_dificuldade      NUMBER(1) NOT NULL,
    dt_config           DATE NOT NULL
);

CREATE UNIQUE INDEX un_config_us_cd_usuario ON
    t_am_config_usuario (
        cd_usuario
    ASC );



ALTER TABLE t_am_config_usuario
    ADD CONSTRAINT ck_configusu_opcmsgv CHECK (
        nr_opcao_receberm = 1
        OR nr_opcao_receberm = 0
    );

ALTER TABLE t_am_config_usuario ADD CONSTRAINT pk_am_config_usuario PRIMARY KEY ( cd_config,
                                                                                  cd_usuario );
alter table T_AM_CONFIG_USUARIO ADD CONSTRAINT CK_AM_CONFIGID CHECK(CD_CONFIG=1);
 

ALTER TABLE t_am_config_usuario
    ADD CONSTRAINT fk_am_config_dificuldade FOREIGN KEY ( cd_dificuldade )
        REFERENCES t_am_dificuldade ( cd_dificuldade );

ALTER TABLE t_am_config_usuario
    ADD CONSTRAINT fk_am_config_lingua FOREIGN KEY ( cd_lingua )
        REFERENCES t_am_lingua ( cd_lingua );

ALTER TABLE t_am_config_usuario
    ADD CONSTRAINT fk_am_config_usuario FOREIGN KEY ( cd_usuario )
        REFERENCES t_am_usuario ( cd_usuario );
COMMIT;
